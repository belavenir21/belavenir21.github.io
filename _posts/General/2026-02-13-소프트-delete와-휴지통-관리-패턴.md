# 소프트 Delete와 휴지통 관리 패턴

trash/restore/empty_trash/hard_delete 등의 변화는 데이터 생명주기와 UX 모두에 큰 영향을 줍니다. 이 주제는 데이터 복구 시나리오, 합산 로직, 영구 삭제를 안전하게 처리하는 전략을 다룹니다.

문자열 정규화와 동의어 매핑

대소문자 표준화, 공백 제거, 한글 표기 차이를 일관된 키로 변환하고, 사전 정의된 동의어를 치환하는 과정을 통해 일관된 비교를 가능하게 하는 원리

데이터 병합 및 복구 로직

활성 항목이 이미 존재하는 경우 수량 등을 합산하는 병합 로직과, 복구 시 기존 데이터와의 충돌 해결 전략의 설계 원리

API 계약과 데이터 일관성

백엔드의 상태 변화가 프런트엔드와 다른 레이어에 어떤 영향을 주는지, 변경 전후의 데이터 모델/형식의 호환성을 어떻게 유지하는지에 대한 원리

```python
def normalize_name(name: str) -> str:
    # 기본 예시: 소문자화 및 공백 제거
    s = name.lower().strip().replace(" ", "")
    return s


def check_match(name1: str, name2: str) -> bool:
    n1 = normalize_name(name1)
    n2 = normalize_name(name2)
    if n1 == n2:
        return True
    synonyms = {
        '달걀': '계란', '계란': '달걀',
        '소고기': '쇠고기', '쇠고기': '소고기',
        '양파': '생양파'
    }
    # 간단한 동의어 매핑 검사
    if (n1 in synonyms and synonyms[n1] == n2) or (n2 in synonyms and synonyms[n2] == n1):
        return True
    # 부분 문자열 매칭
    if n1 in n2 or n2 in n1:
        return True
    return False

# 예시
print(check_match('달걀', '계란'))  # True
print(check_match('소고기', '쇠고기'))  # True
print(check_match('양파', '양파'))  # True
print(check_match('양파', '생양파'))  # True
print(check_match('당근', '감자'))  # False
```
